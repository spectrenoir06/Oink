import 'System'
import 'UnityEngine'
import 'Assembly-CSharp'

function enter(npc)
	npc:startWalking();
	rand = npc:random(1.0, 5.0); 
	npc:wait(rand, "throwFish");
end

function update(npc)
	
end

function throwFish(npc)
	npc:throwFish();
	npc:wait(10, "seekFish");
end

function seekFish(npc)
	npc:changeState("PingouinFindFish");
end
