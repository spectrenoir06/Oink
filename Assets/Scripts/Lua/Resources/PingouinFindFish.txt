import 'System'
import 'UnityEngine'
import 'Assembly-CSharp'

Fish = nil
foundFish = false;

function enter(npc)

end

function update(npc)
	if not foundFish then
		npc:goToClosestFish();
	end
end

function onFindFish(npc, fish)
	foundFish = true;
	Fish = fish;
	npc:playAnimation("dance&eat");
end

function onAnimEnd(npc)
	Debug.Log("ANIM END");
	if Fish then
		Fish:eat();
	end
	npc:changeState("PingouinWalk");
end
